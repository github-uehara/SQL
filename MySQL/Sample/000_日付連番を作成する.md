```
SET
    @start_day = '2023-01-01'; 

SELECT
    @start_day + INTERVAL seq_no DAY AS '年月日' 
FROM
    ( 
        SELECT
            @seq_no := 0 AS seq_no 
        UNION 
        SELECT
            @seq_no := @seq_no + 1 AS seq_no 
        FROM
            information_schema.COLUMNS 
        LIMIT
            10
    ) AS Sample;
```
